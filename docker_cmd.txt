#create private docker network
docker network create --subnet=172.19.0.0/16 iot_network
docker network ls 

#run iot-sensor container using iot-image
docker run --cap-add=NET_ADMIN -it --rm --hostname iot-sensor --net iot_network --ip 172.19.0.11 -v iot_vol:/root/iot_vol --name iot-sensor iot-image bash

#run iot-mqtt container using iot-image
docker run --cap-add=NET_ADMIN -it --rm --hostname iot-mqtt --net iot_network --ip 172.19.0.12 -v iot_vol:/root/iot_vol --name iot-mqtt iot-image bash

#run iot-db container using iot-image
docker run --cap-add=NET_ADMIN -it --rm --hostname iot-db --net iot_network --ip 172.19.0.13 -v iot_vol:/root/iot_vol --name iot-db iot-image bash

#options
--cap-add=NET_ADMIN 
capability add network admin privilege

-it
interactive terminal mode

--rm
remove container after exit

--hostname
container hostname

--net
specify private container network

--ip
assign static IP to container

-v
volumes to mount:container target directory 

--name
assign container name
